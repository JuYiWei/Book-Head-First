2.2 代码

  • Duck
@interface Duck : NSObject

@property (nonatomic, strong) DuckVoiceBehavior *say;
@property (nonatomic, strong) DuckFlyBehavior *fly;

- (void)display;

@end


@implementation Duck

- (void)display {
    NSLog(@"鸭子的标准外观");
}

- (DuckVoiceBehavior *)say {
    if (!_say) {
        _say = [[DuckVoiceBehavior alloc] init];
    }
    return _say;
}

- (DuckFlyBehavior *)fly {
    if (!_fly) {
        _fly = [[DuckFlyBehavior alloc] init];
    }
    return _fly;
}

@end
  • DuckVoiceBehavior
@interface DuckVoiceBehavior : NSObject

- (void)sayGaGa;

- (void)sayZhiZhi;

@end


@implementation DuckVoiceBehavior

- (void)sayGaGa {
    NSLog(@"GaGa");
}

- (void)sayZhiZhi {
    NSLog(@"ZhiZhi");
}

@end
  • DuckFlyBehavior
@interface DuckFlyBehavior : NSObject

- (void)flySlowly;

- (void)flyDisable;

@end


@implementation DuckFlyBehavior

- (void)flySlowly {
    NSLog(@"fly slowly");
}

- (void)flyDisable {
    NSLog(@"can not fly");
}

@end
