第 8 章 模板方法模式

  • 一、基础
  • 二、实践
    • 2.1 需求
    • 2.2 代码

由子类决定如何实现一个算法中的步骤

一、基础

在一个方法汇中定义一个算法的骨架，
而将一些步骤延伸到子类中。
模板方法使得子类可以在不改变算法结构的情况下，
重新定义算法中的某些步骤。

二、实践

2.1 需求

咖啡馆泡咖啡、泡茶流程

迭代
制作果汁

2.2 代码

  • Beverage 父类
@interface Beverage : NSObject

- (void)prepareRecipe;

- (void)boilWater;
- (void)brew;
- (void)pourInCup;
- (void)addCondiments;

@end


@implementation Beverage

// 流程模板
- (void)prepareRecipe {
    [self boilWater];
    [self brew];
    [self pourInCup];
    [self addCondiments];
}

// 详细步骤，子类可以重写
- (void)boilWater {
    NSLog(@"烧热水");
}

- (void)brew {
    NSLog(@"冲泡");
}

- (void)pourInCup {
    NSLog(@"倒进杯子");
}

- (void)addCondiments {
    NSLog(@"加调料");
}

@end
  • Coffee
@implementation Coffee

- (void)brew {
    NSLog(@"泡咖啡");
}

- (void)addCondiments {
    NSLog(@"加奶，加糖");
}

@end
  • 使用
    Coffee *coffee = [[Coffee alloc] init];
    [coffee prepareRecipe];// 调用模板方法
